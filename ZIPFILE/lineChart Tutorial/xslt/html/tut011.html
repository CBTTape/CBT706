Add a background image.

You do this by adding a <style> tag, and a <plotarea> tag under it, and then a <picture> tag under that specifying the name of the image file (or URL). By default, the image is stretched to fit the plot area unless you specify a "location" attribute on the <picture> tag. This is explained later.

Chart xml sourceResulting image
<chart>
    <title>Weather Balloon Tracking</title>

    <series>
        <title>Aquarius</title>
        <datum y="1"/>
        <datum y="3"/>
        <datum y="9"/>
        <datum y="8"/>
        <datum y="7"/>
    </series>

    <series>
        <title>Scorpio</title>
        <datum y2="11"/>
        <datum y2="117"/>
        <datum y2="87"/>
        <datum y2="63"/>
        <datum y2="2"/>
    </series>

    <style>
        <plotarea>
            <picture>background.jpg</picture>
        </plotarea>
    </style>

    <layout>
        <south>
            <legend/>
        </south>
    </layout>

    <axis>
        <y>
            <title>Aquarius Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>15</max>
            </range>
            <interval>
                <major>5</major>
            </interval>
        </y>
        <y2>
            <title>Scorpio Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>150</max>
            </range>
        </y2>
        <x>
            <title>Time Since Launch (days)</title>
        </x>
    </axis>

</chart>
