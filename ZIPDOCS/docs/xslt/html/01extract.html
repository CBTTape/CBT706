How to create an XML file

  • Use any tool or programming language that can create a text file. For example, the following Rexx code can be used to create a simple chart xml file containing random data:

    /*REXX*/
      address TSO
      call Prolog
      call Chart 'Daily CPU Utilization'
      call Epilog
    exit

    Chart: procedure
      parse arg sChartTitle
      call Say '<chart>'
      call Say '  <title>'sChartTitle'</title>'
      call Series 'TSO',10,25
      call Series 'Batch',1,10
      call Series 'DB2',10,100
      call Say '</chart>'
    return

    Series: procedure
      parse arg sSeriesTitle,nPercentFrom,nPercentTo
      call Say '  <series>'
      call Say '    <title>'sSeriesTitle'</title>'
      call Data nPercentFrom,nPercentTo
      call Say '  </series>'
    return

    Data: procedure
      parse arg nFrom,nTo
      do nHour = 8 to 17
        nUtil = random(nFrom,nTo)
        call Say '    <datum x="'nHour'" y="'nUtil'"/>'
      end
    return

    Prolog:
      'ALLOC DD(OUTPUT) DSN(CHART.XML) NEW' /* ...change NEW to SHR after the first run! */
      call Say '<?xml version="1.0" encoding="ebcdic-cp-us"?>'
    return

    Epilog:
      'EXECIO 0 DISKW OUTPUT (FINIS'
      'FREE DD(OUTPUT)'
    return

    Say: procedure
      parse arg sData
      queue sData
      'EXECIO 1 DISKW OUTPUT'
    return
