Change the opacity of the background image

Reducing the background image opacity is a good way to make the plotted data easier to see. Here we reduce the opacity of the background image from 1 down to 0.5 (1 is opaque, 0 is transparent). This will cause the background to punch through the image, so you may want to make sure that the background is white by preceding the <picture> tag with <color>white</color>.

Chart xml sourceResulting image
<chart>
    <title>Weather Balloon Tracking</title>

    <series>
        <title>Aquarius</title>
        <datum y="1"/>
        <datum y="3"/>
        <datum y="9"/>
        <datum y="8"/>
        <datum y="7"/>
    </series>

    <series>
        <title>Scorpio</title>
        <datum y2="11"/>
        <datum y2="117"/>
        <datum y2="87"/>
        <datum y2="63"/>
        <datum y2="2"/>
    </series>

    <style>
        <plotarea>
            <picture opacity="0.5">background.jpg</picture>
        </plotarea>
    </style>

    <layout>
        <south>
            <legend/>
            <labels>
                <rotate>-90</rotate>
                <height>40</height>
            </labels>
        </south>
    </layout>

    <axis>
        <y>
            <title>Aquarius Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>15</max>
            </range>
            <interval>
                <major>5</major>
            </interval>
        </y>
        <y2>
            <title>Scorpio Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>150</max>
            </range>
        </y2>
        <x>
            <title>Time Since Launch</title>
            <label>00:00</label>
            <label>01:00</label>
            <label>02:00</label>
            <label>03:00</label>
            <label>04:00</label>
        </x>
    </axis>

</chart>
