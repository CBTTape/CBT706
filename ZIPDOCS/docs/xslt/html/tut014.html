Increase the height of the X-axis labels area

Now we can turn off the debug option, by either removing the debug attribute from the <chart> tag, or changing it to anything other than "true", and increase the height of the X-axis labels area by adding a <height> tag to the <labels> tag in the <layout> section.

Chart xml sourceResulting image
<chart debug="false">
    <title>Weather Balloon Tracking</title>

    <series>
        <title>Aquarius</title>
        <datum y="1"/>
        <datum y="3"/>
        <datum y="9"/>
        <datum y="8"/>
        <datum y="7"/>
    </series>

    <series>
        <title>Scorpio</title>
        <datum y2="11"/>
        <datum y2="117"/>
        <datum y2="87"/>
        <datum y2="63"/>
        <datum y2="2"/>
    </series>

    <layout>
        <south>
            <legend/>
            <labels>
                <rotate>-90</rotate>
                <height>40</height>
            </labels>
        </south>
    </layout>

    <axis>
        <y>
            <title>Aquarius Height Above Sea Level (km)</title>
            <range>
                <min>0</
min>
                <max>15</max>
            </range>
            <interval>
                <major>5</major>
            </interval>
        </y>
        <y2>
            <title>Scorpio Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>150</max>
            </range>
        </y2>
        <x>
            <title>Time Since Launch</title>
            <label>00:00</label>
            <label>01:00</label>
            <label>02:00</label>
            <label>03:00</label>
            <label>04:00</label>
        </x>
    </axis>

</chart>
