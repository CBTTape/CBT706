Set the axis titles

This time we add <title> tags under the <axis> tag as follows.

Chart xml sourceResulting image
<chart>
    <title>Weather Balloon Tracking</title>

    <series>
        <title>Aquarius</title>
        <datum y="1"/>
        <datum y="3"/>
        <datum y="9"/>
        <datum y="8"/>
        <datum y="7"/>
    </series>

    <series>
        <title>Scorpio</title>
        <datum y2="11"/>
        <datum y2="117"/>
        <datum y2="87"/>
        <datum y2="63"/>
        <datum y2="2"/>
    </series>

    <layout>
        <south>
            <legend/>
        </south>
    </layout>

    <axis>
        <y>
            <title>Aquarius Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>15</max>
            </range>
            <interval>
                <major>5</major>
            </interval>
        </y>
        <y2>
            <title>Scorpio Height Above Sea Level (km)</title>
            <range>
                <min>0</min>
                <max>150</max>
            </range>
        </y2>
        <x>
            <title>Time Since Launch</title>
        </x>
    </axis>

</chart>
